## Jenkins
```
pipeline {
    agent {
        docker {
            image 'mcr.microsoft.com/dotnet/sdk:8.0'
            args '-u root:root' 
        }
    }
    
    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        
        stage('Build .NET Application') {
            steps {
                script {
                    sh '''
                        dotnet publish MyCLassLibrary.csproj -c Release
                    '''
                }
            }
        }
        
        stage('Create NuGet Package') {
            steps {
                script {
                    sh '''
                        dotnet pack MyCLassLibrary.csproj -c Release -o ./packages
                    '''
                }
            }
        }
    }
    
    post {
        success {
            archiveArtifacts artifacts: '**/*.nupkg', followSymlinks: false
        }
        failure {
            echo 'Pipeline failed! Check the logs for details.'
        }
        cleanup {
            cleanWs()
        }
    }
}
```
